Example2: Showing text, original link to http://google.com and lancelot link to http://ttlocal.info 

		$('.lancelot2').lancelot({
			hoverTime: 3000,

			atext: "hover here!",

Example3: Using span in the generated link. 

		$('.lancelot3').lancelot({
			hoverTime: 1000,
			aclass: "lancelotGo3",
			atext: "go",
			speed: "slow",
			element: "span",	//< span > instead of < a >
			linkAction: "_blank"	//new window
		});
		<a href="http://github.com/codexico/jquery-plugin-lancelot-autolink" class="lancelot3">Fork me!</a>

		--

		Result:
		<a class="lancelot3" href="http://github.com/codexico/jquery-plugin-lancelot-autolink">Fork me!
			<span class="lancelotGo3" style="display: none;"></span>
		</a>


		CSS:
		.lancelot3{
			background-color: yellow;
		}
		.lancelotGo3{
			background: url("images/lancelot-icon.png") no-repeat scroll right center transparent;
			padding:0 13px 0 6px;
		}
Fork me!


Example4: Using a function to generate the lancelot link. 

		var googleSearch = function(obj){
			//obj is a reference to each $('.lancelot4')
			var search = obj.text().replace(/^#/,'');
			return "http://www.google.com/search?q="+search
		};
		$('.lancelot4').lancelot({
			hoverTime: 3000,
			aclass: "lancelotGo4",
			atext: "",
			show: "false",
			speed: "fast",
			linkAction: "location",
			alink: googleSearch			//passing function!
		});
		<a href="http://jquery.com" class="lancelot4">lancelot</a>

		--

		Result:
		<a class="lancelot4" href="http://jquery.com">lancelot
			<a title="go in 2s" class="lancelotGo4" href="http://www.google.com/search?q=lancelot" style="display: inline;"></a>
		</a>


		CSS:
		.lancelot4{
			border: 1px solid green;
		}
		.lancelotGo4{
			background:url("images/external-link-ltr-icon.png") no-repeat scroll left center transparent;
			padding:0 0 0 13px;
		}
